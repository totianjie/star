entry    （入口配置）有三种 写法
    entry: './src/a.js'
    entry: ['./src/a.js', './src/b.js'] // ==> 按钮顺序依次打包
    entry: {
        a: './src/a.js,
        b: './src/b.js'
    }
output    （出口配置） 两种写法
    output: {    // 打包到一个js文件中
        path: path.resolve(__dirname, 'dist'),
        filename: 'buidle.js'     
    }
    output: {    // 跟据entry中的json格式打包出多个文件
        path: path.resolve(__dirname, 'dist'),
        filename: '[name].buidle.js'     
    }
    
output: {
    path: path.resolve(__dirname, 'dist'), // path一定是要绝对路径    ==>    __dirname + '/dist'   (这样写也是一样的，同理)
    filename: 'bundle.js'
}

plugins   插件

html-webpack-plugin    (生成页面, 自动引入打包好的js文件)
    npmjs地址：https://www.npmjs.com/package/html-webpack-plugin
    博客地址：https://www.cnblogs.com/wonyun/p/6030090.html

    注意：    此模块依赖webpack和webpack-cli        cnpm install webpack webpack-cli -D
    
    下载：cnpm install html-webpack-plugin --save-dev
    引入：let HtmlWebpackPlugin = require('html-webpack-plugin');
   
   使用：
let path = require('path');
let HtmlWebpackPlugin = require('html-webpack-plugin');

module.exports = {
    entry: {      
        index: './src/index.js',
        index2: './src/index2.js'
    },
    output: {
        path: path.resolve(__dirname, 'dist'),
        filename: 'bundle.js'
    },
    plugins: [
        new HtmlWebpackPlugin({
            title: '你好 htmlWebpackPlugin', // 使用<title><%= htmlWebpackPlugin.options.title%></title>
            filename: 'index.html', // 默认生成index.html 如果要生成多个页面要指定这个名字， new HhtmlWebpackPlugin() 要调用多次
            template: './src/index.html', // 相对于webpack.config.js的路径
            hash: true, // 每次打包清缓存, ？后面加了一个随机数
            minify: {     
                collapseWhitespace: true, // 压缩
                removeAttributeQuotes: true, // 删除html双引号
            },
            inject: true, // 默认true/'body' | 'head' | false 所有静态资源不会被引入
            chunks: ['index2'], // 默认是把全部打包的js引入 ，这里是指定需要引入的js文件（主要是多页面时会用到），相对于是entry里的名字
        })
    ]
};

devServer
    cnpmjs地址：https://www.npmjs.com/package/webpack-dev-server
    下载：cnpm install  webpack-dev-server -D
    运行时如果提示少了什么模块就直接删除node_modules重新下载
    注意：要想devServer中的hot生效一定要引入webpack模块，在plugins里面调用一下new Webpack.HotModuleReplacementPlugin(), // devServer中的hot生效
    使用：webpack.config.js
// devServer 配制
let path = require('path');
let HtmlWebpackPlugin = require('html-webpack-plugin');
let CleanWebpackPlugin = require('clean-webpack-plugin');
let Webpack = require('webpack');
module.exports = {
    entry: {
        index: './src/index.js'
    },
    output: {
        path: path.resolve(__dirname, 'dist'),
        filename: '[name].bundle.js'
    },
    devServer: {
        contentBase: __dirname + 'dist', // 设置服务器访问的基本目录 跟output的path是相对应的
        host: 'localhost', // 服务器ip地址
        port: 8081, // 商品
        open: true, // 自动打开浏览器
        hot: true, // 热更新
    },
    plugins: [
        new Webpack.HotModuleReplacementPlugin(), // devServer中的hot生效
        new CleanWebpackPlugin(['dist']),
        new HtmlWebpackPlugin({
            title: '你好 htmlWebpackPlugin', // 使用<title><%= htmlWebpackPlugin.options.title%></title>
            filename: 'index.html', // 默认生成index.html 如果要生成多个页面要指定这个名字， new HhtmlWebpackPlugin() 要调用多次
            template: './src/index.html', // 相对于webpack.config.js的路径
            hash: true, // 每次打包清缓存, ？后面加了一个随机数
            minify: {
                collapseWhitespace: true, // 压缩
                removeAttributeQuotes: true, // 删除html双引号
            },
            inject: true, // 默认true/'body' | 'head' | false 所有静态资源不会被引入
            chunks: ['index'], // 默认是把全部打包的js引入 这里是指定需要引入的js文件（主要是多页面时会用到），相对于是entry里的名字
        })
    ]
};

loaders  加载器，转换器

处理 css文件
    style-loader    css-loader
    cnpm install style-loader css-loader -D
   使用：
 index.js入口文件      import './css/common.css'
module: {
    rules: [
        {
            test: /\.css$/,
            use: ['style-loader', 'css-loader']
        }
    ]
},

第一种 写法：loader: ['style-loader', 'css-loader']
第二种 写法： use: ['style-loader', 'css-loader']
第三种 写法：
    use: [
        {loader: 'style-loader'},
        {loader: 'css-loader'}
    ]
第一种和第二种写都是等价的，如果loader需要配制的话就用第三种方法,平时三种方法都是可以用的
分离css  extract-text-webpack-plugin
    npmjs地址： https://www.npmjs.com/package/extract-text-webpack-plugin
   下载： cnpm install extract-text-webpack-plugin -D
            目前webpack4.x使用extract-text-webpack-plugin会报错 因为插件的更新没有跟上，要使用最新的插件才可以
            cnpm install extract-text-webpack-plugin@next -D    下载最新的插件
    引入：const ExtractTextPlugin = require('extract-text-webpack-plugin');
    
    module: {
        rules: [
            {   
                test: /\.css$/,
                // use: [ // 不需要分离css这样写
                //     {loader: 'style-loader'},
                //     {loader: 'css-loader'}
                // ],
                use: ExtractTextPlugin.extract({    // 通过ExtractTextPlugin把css分离出去
                    fallback: 'style-loader',
                    use: 'css-loader',
                    publicPath: '../', // 处理背景图的问题,以目前现在的配制图片的目录在 ../  上一级，这个按实际情况  **这个一定要注意
                })
            },
            {
                test: /\.(jpg|png|gif)$/,
                use: [
                    {
                        loader: 'url-loader',
                        options: {
                            limit: 5000,
                            outputPath: 'images', // 打包出去的文件目录， 不配的话默认是把全部图片放根目录下  **这个一定要注意
                        }
                    }
                ]
            }
        ]
    },
    plugins: [
        new ExtractTextPlugin('css/index.css')     // 括号里面放的是提取出去后的路径
    ],


分离css别一个插件    mini-css-extract-plugin   只为提取css设计    目前(2018,3,31)还不好用（路径问题处理 不了，还没有这个配制，publicPath没有）

下载：npm install --save-dev mini-css-extract-plugin
引入：const MiniCssExtractPlugin = require("mini-css-extract-plugin");
使用：

module.exports = {
  plugins: [
    new MiniCssExtractPlugin({
      filename: "[name].css",  // 提取出去后的名字  可以直接写成index.css
    })
  ],
  module: {
    rules: [
      {
        test: /\.css$/,
        use: [
          MiniCssExtractPlugin.loader,
          "css-loader"
        ]
      }
    ]
  }
}

如何引入第三方库（插件）  要单独打包出去目前推荐第三种方法
    1，cnpm install jquery -S
         直接在要用的js文件中    import $ from 'jquery';    // 这种方式引：只要你引入了你用不用都会打包进去
  2, 在webpack.config.js  里的plugins里面引入    // 这样引的好处是如果你在自己的项目中没有使用，build打包是不会打进去的
    
new webpack.ProvidePlugin({
    $: 'jquery'
})
 
3, 在webpack.config.js  里的entry里引入        // 这样引入的方式就比较的好只要在output里面配制就可以单独打包出去  
entry: {
    jquery: './static/jquery.min.js',
    index: './src/index.js',
    a: './src/a.js'
},
output: {
    path: path.resolve(__dirname, 'dist'),
    filename: 'js/[name].bunld.js'        // 这里配制了
},

    4，在webpack.config.js里配制    optimization    目前2018.4.1为此这种方式是可以成功了也在index.html里自动引入了，但提示$ is not defined
entry: {
    index: './src/index.js',
    jquery: 'jquery'
},
output: {
    path: path.resolve(__dirname, 'dist'),
    filename: 'bunld.js'
},

optimization: {
    splitChunks: {
        cacheGroups: {
            aaa: {
                chunks: 'initial',
                name: 'jquery',
                enforce: true
            }
        }
    }
}
